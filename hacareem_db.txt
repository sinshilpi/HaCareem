CREATE TABLE Partner
(
id INT NOT NULL AUTO_INCREMENT,
partnerName varchar(50),
PRIMARY KEY(id)
);
CREATE TABLE Vehicle
(
id INT NOT NULL AUTO_INCREMENT,
size int,
type Varchar(20),
PRIMARY KEY(id)
);
CREATE TABLE Item
(
id INT NOT NULL AUTO_INCREMENT,
item_name varchar(50),
item_weight int,
item_height int,
item_length int,
item_breadth int,
fragile bool,
PRIMARY KEY(id)
);
CREATE TABLE Hub
(
id INT NOT NULL AUTO_INCREMENT,
state varchar(10),
city varchar(10),
PRIMARY KEY(id)
);
CREATE TABLE Company
(
id INT NOT NULL AUTO_INCREMENT,
comp_name varchar(10),
PRIMARY KEY(id)
);    
CREATE TABLE Vehicle_Instance
(
id INT NOT NULL AUTO_INCREMENT,
vehicle_id int,
source_hub_id int,
destination_hub_id int,
current_hub int,
partner_id int,
PRIMARY KEY(id)
);    
CREATE TABLE Order
(
id INT NOT NULL AUTO_INCREMENT,
source_hub_id int,
destination_hub_id int,
comp_id int,
order_date Date,
price double,
PRIMARY KEY(id)
);    
CREATE TABLE Order_Item
(
id INT NOT NULL AUTO_INCREMENT,
order_id int,
item_id int,
price double,
PRIMARY KEY(id)
);    
CREATE TABLE Order_Track
(
id INT NOT NULL AUTO_INCREMENT,
order_id int,
hub_id int,
vehicle_id int,
status int,
PRIMARY KEY(id)
);    
CREATE TABLE Payment
(
id INT NOT NULL AUTO_INCREMENT,
order_id int,
bank_details varchar(20),
company_id int,
amount_paid double,
PRIMARY KEY(id)
);
